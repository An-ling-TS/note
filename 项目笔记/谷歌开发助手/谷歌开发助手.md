#### 进阶——样式优化

##### 思路

> 当节点存在子节点时，跳向子节点
>
> 当节点不存在非文本子节点时，获取节点的width与height属性，如果节点存在class类名，则使用获取到的width和height属性替换掉outerHTML中的class类名,然后返回outerHTML
>
> 当子节点处理完毕时，使用处理后的子节点outerHTML代替当前节点的innerHTML
>
> 根节点处理完毕后，返回根节点的outerHTML
>
> 之后的步骤与上一节表单生成步骤相似

export function replaceClassWithStyle(domNode: any) {
    let resp = ''
    if (domNode.children) {
        for (let i = 0; i < domNode.children.length; i++) {
            resp += replaceClassWithStyle(domNode.children[i])
        }
    }
    //保留宽度和高度属性
    let styleStr=`"
    width:${window.getComputedStyle(domNode, null)['width']};
    height:${window.getComputedStyle(domNode, null)['height']};
    "`.replace(/\s/g, '')
    //获取包含当前节点在内的HTML字符串
    let temp = domNode.outerHTML.replace(/\n*/g, '')
    if (resp) {
        //如果存在非文本子节点，则使用replaceClassWithStyle后的子节点代替原来的子节点
        //temp = temp.replace(/>.*</g, `>${resp}<`)
        temp = temp.replace(domNode.innerHTML.replace(/\n*/g, ''), `${resp}`)
    } else {
        //不存在非文本子节点
        if (!(
            (/\/>/).test(temp)
            ||
            (/<\//).test(temp)
        )) {
            //当为自闭合标签或者存在文本子节点时
            temp = temp.replace(/>/, '/>')
        }
    }
    //使用style属性字符串代替class类名字符串
    resp = temp.replace(/class=".*?"/, styleStr.length === 2 ? '' : `style=${styleStr}`)
    return resp
}

export function isFalse(value: any) {
    if (value === undefined) return true;
    const type = typeof value;
    if (type === 'object') {
        if (value === null) return true;
        if (Array.isArray(value) && value.length === 0) return true;

//获取props不同真值构成的新数组
export function distinct(array: Array<any>, props?: string, backward = true) {
    //debugger
    if (!props) {
        return Array.from(new Set(array))
    }
    let res = new Array();//存放结果
    let baseTypeData = new Set();//存放基础类型
    let propsData = new Array();
    let indexData = new Array();
    array.forEach((i, index) => {
        console.log('i', i)
        if (typeof i === 'object' && !Array.isArray(i)) {
            if (isFalse(i[props])) {
                propsData.push(i[props]);
                indexData.push(index)
            } else {
                const propsIndex = propsData.indexOf(i[props])
                //如果propsData中存在，说明前面的遍历已经遇到了.array[i]是props重复项
                if (propsIndex > -1) {
                    //如果是向后保存，即props相同时保存靠后的对象
                    if (backward) {
                        //更新indexData
                        indexData.splice(propsIndex, 1);
                        indexData.push(index);
                        //保存propsData与indexData顺序一致
                        propsData.splice(propsIndex, 1);
                        propsData.push(i[props])
                    }
                } else {
                    propsData.push(i[props]);
                    indexData.push(index)
                }
            }
        }
    })
    array.forEach((i, index) => {
        if (typeof i === 'object' && indexData.indexOf(index) > -1) {
            res.push(i)
        } else if (typeof i !== 'object' && !baseTypeData.has(i)) {
            res.push(i);
            baseTypeData.add(i)
        }
    })
    return res;
}

//过滤掉对象的空属性
export function filterNullValueObject(obj: any) {
    if (typeof obj !== 'object' || Array.isArray(obj)) {
        return { }
    }
    interface Obj {
        [key: string]: any
    }
    let newObj: Obj = { }
    Object.keys(obj).forEach((item: string) => {
        if (obj[item] === 0 || Boolean(obj[item])) {
            if ((!Array.isArray(obj[item]) || (Array.isArray(obj[item]) && obj[item].length > 0))) {
                newObj[item] = obj[item]
            }
        }
    })
    return newObj;
}
